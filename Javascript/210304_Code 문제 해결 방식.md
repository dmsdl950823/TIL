``` js
  // 사용법
  function factorialWrap (nodewrap) {
    const result = []
    factorial(nodewrap)

    function factorial (nodes) {
      return nodes.map(node => {
        if (node.children) {
          const nodeCopy = { ...node }
          delete nodeCopy.children
          result.push(nodeCopy)
          return factorial(node.children)
        } else result.push(node)
      })
    }
    return result
  }

  function root (nodes) {
    // output
    return factorialWrap(nodes)
  }

  root(data)
```
